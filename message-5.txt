local function getProfileThumbnail()
    local success, content = pcall(function()
        local userId = player.UserId
        local thumbType = Enum.ThumbnailType.HeadShot
        local thumbSize = Enum.ThumbnailSize.Size100x100
        return game:GetService("Players"):GetUserThumbnailAsync(userId, thumbType, thumbSize)
    end)
    return success and content or "rbxassetid://0"
end

local function isInside(indicator, target)
    if not indicator or not target then return false end
    local iPos = indicator.AbsolutePosition + (indicator.AbsoluteSize / 2)
    local tPos = target.AbsolutePosition
    local tSize = target.AbsoluteSize
    
    return iPos.X >= tPos.X and iPos.X <= (tPos.X + tSize.X) and
           iPos.Y >= tPos.Y and iPos.Y <= (tPos.Y + tSize.Y)
end

    toggleButton.BorderSizePixel = 0

    toggleButton.Size = UDim2.new(0, 50, 0, 50)
    toggleButton.Image = "rbxassetid://81617068776053" -- List/Menu icon

    stroke.Color = Color3.fromRGB(60, 60, 60)
    stroke.Parent = toggleButton

        if window then
            window:Minimize()
        end

    task.spawn(function()
        local sidebar
        local start = tick()
        
        -- Enhanced search for sidebar
        while not sidebar and tick() - start < 15 do
            if window and window.Root then
                -- Try finding by common name or class
                sidebar = window.Root:FindFirstChild("Sidebar", true) 
                
                if not sidebar then
                    -- Fallback: Look for any ScrollingFrame that might be the sidebar
                    for _, v in pairs(window.Root:GetDescendants()) do
                        if v:IsA("ScrollingFrame") and v.Name == "Sidebar" then
                            sidebar = v
                            break
                        end
                    end
                end
            end
            if not sidebar then task.wait(0.5) end
        end

        if sidebar then
            if sidebar:FindFirstChild("UserProfile") then return end

            local profileFrame = Instance.new("Frame")
            profileFrame.Name = "UserProfile"
            profileFrame.Parent = sidebar
            profileFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
            profileFrame.BackgroundTransparency = 0.1
            profileFrame.BorderSizePixel = 0
            -- Position it at the bottom of the sidebar
            profileFrame.Position = UDim2.new(0, 10, 1, -70)
            profileFrame.Size = UDim2.new(1, -20, 0, 60)
            profileFrame.ZIndex = 500
            
            local pCorner = Instance.new("UICorner")
            pCorner.CornerRadius = UDim.new(0, 12)
            pCorner.Parent = profileFrame

            local pStroke = Instance.new("UIStroke")
            pStroke.Thickness = 1
            pStroke.Color = Color3.fromRGB(100, 100, 100)
            pStroke.Parent = profileFrame

            local profileImage = Instance.new("ImageLabel")
            profileImage.Name = "Avatar"
            profileImage.Parent = profileFrame
            profileImage.BackgroundTransparency = 1
            profileImage.Position = UDim2.new(0, 10, 0, 10)

-- ฟังก์ชันสำหรับส่งคำสั่งตกปลา
local function startFishing()
    local remotes = ReplicatedStorage:WaitForChild("VerdantRemotes")
    remotes:WaitForChild("VDT_Fishing.Windup"):FireServer()
    remotes:WaitForChild("VDT_ToolHandler.Activated"):FireServer()
    
    local args = { 0.5 } -- ต้องเป็น Table เพื่อใช้ unpack
    remotes:WaitForChild("VDT_Fishing.RequestCast"):FireServer(unpack(args))
end

-- รอโหลด UI ของมินิเกม

local function m1()
    local vim = game:GetService("VirtualInputManager")
    vim:SendMouseButtonEvent(0, 0, 0, true, game, 0) -- กดคลิกซ้าย
    task.wait(0.1)
    vim:SendMouseButtonEvent(0, 0, 0, false, game, 0) -- ปล่อยคลิกซ้าย
    
end

local function rod()
    if not pgui:FindFirstChild("ReelBar")  then
        startFishing()
        task.wait(1)
    
    local strengthBar = pgui:FindFirstChild("StrengthBar")
    if strengthBar and strengthBar.Enabled then
        local sc1 = strengthBar.Holder.ReelGame.Holder.Scale
        if sc1.Size.Y.Scale >= 0.7 then
            m1()
            task.wait(0.2)
        end
    
        
    end
    end
end

local function reel()
    if not pgui:FindFirstChild("StrengthBar") then
    local reelBar = pgui:FindFirstChild("ReelBar")
    if reelBar and reelBar.Enabled then
        local ho1 = reelBar.Holder.ReelGame.Holder
        local ho2 = ho1.OuterZone
        local ho3 = ho1.YourBar
        if ho3.Position.Y.Scale >= ho2.Position.Y.Scale then
            task.wait(0.1)
            m1()
             task.wait(0.3)
        end
    end
    end
end
local function locknpc()
    local b1 =game:GetService("Players").LocalPlayer.Backpack
    for _, v in ipairs(b1:GetChildren()) do
        local rate = v:GetAttribute("Rarity")
        local id  = v:GetAttribute("_id")
        if rate == "Mythic" or rate == "Legendary" then
           ReplicatedStorage:WaitForChild("VerdantRemotes"):WaitForChild("VDT_Inventory.SetLocked"):FireServer(id,true)
        else
            local gay = {
                0.7
            }
            ReplicatedStorage:WaitForChild("VerdantRemotes"):WaitForChild("VDT_Sell.SellAll"):FireServer(unpack(gay))
        end
    end
end
